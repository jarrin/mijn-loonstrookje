STAP 3: MAAK docker-compose.yml IN JE ROOT VAN JE PROJECT AAN EN ZET DIT ER IN: 

services:
  db:
    image: mysql:8.0
    container_name: mijnloonstrookje_db
    restart: always
    environment:
      MYSQL_DATABASE: mijnloonstrookje
      MYSQL_ALLOW_EMPTY_PASSWORD: "yes"
    ports:
      - "3306:3306"
    volumes:
      - db_data:/var/lib/mysql

  phpmyadmin:
    image: phpmyadmin:latest
    container_name: mijnloonstrookje_pma
    restart: always
    environment:
      PMA_HOST: db
      PMA_PORT: 3306
      UPLOAD_LIMIT: 64M
    ports:
      - "8080:80"
    depends_on:
      - db

volumes:
  db_data:
